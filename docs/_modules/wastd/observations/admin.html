[docs]class ObservationAdminMixin(VersionAdmin, admin.ModelAdmin):
    """Admin mixin for Observation models."""

    save_on_top = True
    date_hierarchy = 'encounter__when'
    LIST_FIRST = (
        'pk',
        'area',
        'site',
        'latitude',
        'longitude',
        'date',
    )
    LIST_LAST = (
        'encounter_link',
        'encounter_status',
    )
    LIST_FILTER = (
        ('encounter__area', AreaFilter),
        ('encounter__site', SiteFilter),
        'encounter__status',
        'encounter__encounter_type',
    )
    search_fields = ('comments', )
    readonly_fields = ('encounter', )

    area = forms.ChoiceField(
        widget=ModelSelect2Widget(
            model=Area,
            search_fields=["name__icontains", ]
        )
    )
    site = forms.ChoiceField(
        widget=ModelSelect2Widget(
            model=Area,
            search_fields=["name__icontains", ]
        )
    )
    formfield_overrides = FORMFIELD_OVERRIDES
